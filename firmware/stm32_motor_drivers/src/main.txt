void rebootToDFU() {
    __disable_irq();

    // Stop SysTick
    SysTick->CTRL = 0;

    // Deinit HAL (shuts down peripherals & interrupts)
    HAL_RCC_DeInit();
    HAL_DeInit();

    // ---- Disable USB (STM32G431 version) ----

    // Disable USB clock
    RCC->APB1ENR1 &= ~RCC_APB1ENR1_USBEN;

    // Reset USB peripheral
    RCC->APB1RSTR1 |=  RCC_APB1RSTR1_USBRST;
    RCC->APB1RSTR1 &= ~RCC_APB1RSTR1_USBRST;

    // Reset USB pins PA11/PA12 to analog
    __HAL_RCC_GPIOA_CLK_ENABLE();
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = GPIO_PIN_11 | GPIO_PIN_12;
    GPIO_InitStruct.Mode = GPIO_MODE_ANALOG;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);

    // Remap system memory to 0x00000000
    SYSCFG->MEMRMP = SYSCFG_MEMRMP_MEM_MODE_0;

    // System memory bootloader start
    uint32_t bootAddr = 0x1FFF0000;

    // Load stack pointer and reset vector
    uint32_t sp = *(__IO uint32_t*)(bootAddr);
    uint32_t rv = *(__IO uint32_t*)(bootAddr + 4);

    // Set main stack pointer
    __set_MSP(sp);

    // Jump to bootloader
    ((void (*)(void))rv)();
}

void float2bytesMSB(float val, uint8_t *bytes) {
    // Copy the raw bits of val into an integer so can perform
    // bitwise operations
    uint32_t valAsInt;
    memcpy(&valAsInt, &val, sizeof(float));

    // Extract the bytes in big-endian (MSB-first) order
    bytes[0] = (valAsInt >> 24) & 0xFF;
    bytes[1] = (valAsInt >> 16) & 0xFF;
    bytes[2] = (valAsInt >> 8)  & 0xFF;
    bytes[3] = (valAsInt >> 0)  & 0xFF;
}

float bytes2floatMSB(const uint8_t *bytes) {
    // Readt the bytes in big-endian (MSB-first) order
    uint32_t valAsInt = 
        ((uint32_t)bytes[0] << 24) |
        ((uint32_t)bytes[1] << 16) |
        ((uint32_t)bytes[2] << 8)  |
        ((uint32_t)bytes[3]);

    // Copy the raw bits to a float and return
    float val;
    memcpy(&val, &valAsInt, sizeof(float));

    return val;
}

void setupFOC() {
    // initialise magnetic sensor hardware
    sensor.init(&SPI_1);
    // link the motor to the sensor
    motor.linkSensor(&sensor);

    driver.voltage_power_supply = PSU_voltage;
    driver.voltage_limit = voltage_limit;

    driver.init(&SPI_1);

    // small delay to allow SPI response
    delay(10);

    // Set the buck output tot 5V
    driver.setBuckVoltage(VB_5V);
    // Wait some time for everything to stabilise
    delay(500);

    // It's advised to have at least 400ns between read and write calls
    driver.setOCPLevel(Curr_16A);
    delayMicroseconds(1);
    driver.setOCPMode(AutoRetry_Fault);
    delayMicroseconds(1);
    driver.setOCPRetryTime(Retry500ms);
    delayMicroseconds(1);
    driver.setSlew(Slew_200Vus);
    delayMicroseconds(1);
    driver.setPWM100Frequency(FREQ_20KHz);
    delayMicroseconds(1);
    //driver.setCurrentSenseGain(Gain_0V15); // Set the current sense gain to 150 mV/A
    driver.setCurrentSenseGain(Gain_0V25); // Set the current sense gain to 600 mV/A
    //driver.setCurrentSenseGain(Gain_0V375); // Set the current sense gain to 1200 mV/A
    delayMicroseconds(1);
    // Serial.print("Current gain:");
    // Serial.println("here 1");
    // Serial.println(driver.getCurrentSenseGain()); 
    // Serial.println("here 2");
    // delayMicroseconds(1);
    //driver.setPWMMode(PWM6_Mode);
    driver.setPWMMode(PWM6_Mode);
    delayMicroseconds(1);

    Serial.println("Init complete...");

    delay(100);
    printDRV8316Status();


    motor.current_limit = 2.0;   // [Amps]

    current_sense.linkDriver(&driver);

    // Link driver to the motor
    motor.linkDriver(&driver);

    // aligning voltage 
    motor.voltage_sensor_align = 3;
    // choose FOC modulation 
    motor.foc_modulation = FOCModulationType::SinePWM;

    motor.torque_controller = TorqueControlType::foc_current;
    //motor.torque_controller = TorqueControlType::voltage;
    
    // // // set motion control loop to be used
    motor.controller = MotionControlType::torque;

    
    // // comment out if not needed
    motor.useMonitoring(Serial);

}

void trySendVelocity() {

      uint8_t floatData[4]; // Stores a single float (4 bytes)

      float shaftVel = motor.shaftVelocity();

      CANFDMessage message ;
      message.id = SHAFT_VEL_ADDR ;              // CAN id       
      message.ext = false;              // false = Standard (11-bit) identifier (0x000 to 0x7FF), true = Extended (29-bit) identifier (0x00000000 to 0x1FFFFFFF)

void loop() {

  // --------------------------------------------------------- //
  //                    DFU Reset Triggers                     //
  // --------------------------------------------------------- //

  // Trigger dfu mode if button pressed
  if (digitalRead(USER_BTN) == LOW) {
    rebootToDFU();
  }

  // Trigger dfu if "dfu" received from Serial
  if (Serial.available()) {
        String cmd = Serial.readStringUntil('\n');
        cmd.trim();

        if (cmd == "dfu") {
            Serial.println("Entering DFU mode...");
            delay(10);        // let USB send the message
            rebootToDFU();
        }
    }

  // --------------------------------------------------------- //
  //                         SimpleFOC                         //
  // --------------------------------------------------------- //

  // Run the FOC loop commands
  motor.loopFOC();
  motor.move(target_current);

  // --------------------------------------------------------- //
  //                     Send CAN Messages                     //
  // --------------------------------------------------------- //

  // Send the telemetry at 0.1Hz (once every 10 seconds)
  trySendTelemetry();

  // Send the rotational velocity at 250Hz
  trySendVelocity();

  // --------------------------------------------------------- //
  //                  Receive any CAN messages                 //
  // --------------------------------------------------------- //

  CANFDMessage messageFD ;
  if (fdcan1.receiveFD0 (messageFD)) {
    //gReceivedCount += 1 ;

    //ledState = !ledState;
    //digitalWrite(LED_BUILTIN, ledState);
    
    if (messageFD.id == CAN_ADDR) {
        // TODO - Need to add validation, proper filtering, etc.
        uint8_t received_float_data[4];

        memcpy(received_float_data, messageFD.data, 4);

        target_current = bytes2floatMSB(received_float_data);

        //Serial.printf("Received -> Current = %+.4f A (%x %x %x %x) from -> 0x%X \n", target_current, received_float_data[0], received_float_data[1], received_float_data[2], received_float_data[3], messageFD.id);
        Serial.print("Received");
        constrain(target_current, -1, 1);

        Serial.print("Target current: ");
        Serial.print(target_current);
        Serial.println("A");
    }
    // Serial.print ("fdcan1 received: ") ;
    // Serial.println (gReceivedCount) ;
  }

  // --------------------------------------------------------- //
  //                       End of Loop()                       //
  // --------------------------------------------------------- //
}
